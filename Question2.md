def combinations(arr, r):
    if r == 0:
        return [[]]
    if len(arr) == 0:
        return []
    return combinations(arr[1:], r - 1) + [comb + [arr[0]] for comb in combinations(arr[1:], r)]

def permutations(arr, r):
    if r == 0:
        return [[]]
    return [arr[i:i+1] + p for i in range(len(arr)) for p in permutations(arr[:i] + arr[i+1:], r - 1)]
